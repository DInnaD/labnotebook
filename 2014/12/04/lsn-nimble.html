Lsn Nimble

some sample data

library(sde)
library(nimble)
set.seed(123)
d <- expression(0.5 * (10-x))
s <- expression(1) 
data <- as.data.frame(sde.sim(X0=6,drift=d, sigma=s, T=20, N=100))
## sigma.x not provided, attempting symbolic derivation.
plot(data)

LSN version

lsn <- modelCode({
   theta ~ dunif(1e-10, 100.0)
   sigma ~ dunif(1e-10, 100.0)
       m ~ dunif(-10, 10)
    x[1] ~ dunif(0, 100)
    y[1] ~ dunif(1e-10, 100.0) 

  for(i in 1:(N-1)){
    mu_x[i] <- x[i] + y[i] * (theta - x[i]) 
    x[i+1] ~ dnorm(mu_t[i], sd = sigma) 
    y[i+1] <- y[i] + m * t[i]
  }
})

Nimble parameters. Note we’ve made time explicit, we’ll assume uniform spacing here:

constants <- list(N = length(data$x), t = 1:length(data$x))
inits <- list(theta = 6, m = 0, sigma = 1, y = rep(1,constants$N))

and here we go as before:

Rmodel <- nimbleModel(code = lsn, 
                      constants = constants, 
                      data = data, 
                      inits = inits)
Cmodel <- compileNimble(Rmodel)
mcmcspec <- MCMCspec(Rmodel, print=TRUE)
## [1] end sampler;   targetNode: theta
## [2] RW sampler;   targetNode: sigma,  adaptive: TRUE,  adaptInterval: 200,  scale: 1
## [3] end sampler;   targetNode: m
## [4] end sampler;   targetNode: y[1]
Rmcmc <- buildMCMC(mcmcspec)
Cmcmc <- compileNimble(Rmcmc, project = Cmodel)
Cmcmc(1e5)
## NULL

and examine results

samples <- as.data.frame(as.matrix(nfVar(Cmcmc, 'mvSamples')))
mean(samples$theta)
## [1] 50.0709
mean(samples$m)
## [1] -0.009911516
mean(samples$sigma)
## [1] 1
plot(samples[ , 'm'], type = 'l', xlab = 'iteration', ylab = 'm')
plot(samples[ , 'sigma'], type = 'l', xlab = 'iteration', ylab = expression(sigma))
plot(samples[ , 'theta'], type = 'l', xlab = 'iteration', ylab = expression(theta))
plot(samples[ , 'm'], samples[ , 'sigma'], xlab = 'm', ylab = expression(simga))
